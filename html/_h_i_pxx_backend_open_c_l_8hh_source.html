126  OpenCLFunctionInfoMap &func_info_map) {
127  ocl_kernel = cl_kernel;
128 
129  int err = 0;
130  name = ocl_kernel.getInfo<CL_KERNEL_FUNCTION_NAME>(&err);
131  if (err != CL_SUCCESS) {
132  logError("clGetKernelInfo(CL_KERNEL_FUNCTION_NAME) failed: {}\n", err);
133  }
134 
135  auto it = func_info_map.find(name);
136  assert(it != func_info_map.end());
137  func_info = it->second;
138 
139  // TODO attributes
140  cl_uint NumArgs = ocl_kernel.getInfo<CL_KERNEL_NUM_ARGS>(&err);
141  if (err != CL_SUCCESS) {
142  logError("clGetKernelInfo(CL_KERNEL_NUM_ARGS) failed: {}\n", err);
143  }
144 
145  assert(func_info->ArgTypeInfo.size() == NumArgs);
146 
147  if (NumArgs > 0) {
148  logDebug("Kernel {} numArgs: {} \n", name, NumArgs);
149  logDebug(" RET_TYPE: {} {} {}\n", func_info->retTypeInfo.size,
150  (unsigned)func_info->retTypeInfo.space,
151  (unsigned)func_info->retTypeInfo.type);
152  for (auto &argty : func_info->ArgTypeInfo) {
153  logDebug(" ARG: SIZE {} SPACE {} TYPE {}\n", argty.size,
154  (unsigned)argty.space, (unsigned)argty.type);
155  TotalArgSize += argty.size;
156  }
157  }
158  }
159 
160  OCLFuncInfo *get_func_info() const { return func_info; }
161  std::string get_name() { return name; }
162  cl::Kernel get() const { return ocl_kernel; }
163  size_t getTotalArgSize() const { return TotalArgSize; };
