85 bool open(const char* fileName)
86 {
87 size_t size;
88 char* str;
89 // Open file stream
90 std::fstream f(fileName, (std::fstream::in | std::fstream::binary));
91 // Check if we have opened file stream
92 if (f.is_open())
93 {
94 size_t sizeFile;
95 // Find the stream size
96 f.seekg(0, std::fstream::end);
97 size = sizeFile = (size_t)f.tellg();
98 f.seekg(0, std::fstream::beg);
99 str = new char[size + 1];
100 if (!str)
101 {
102 f.close();
103 return false;
104 }
105 // Read file
106 f.read(str, sizeFile);
107 f.close();
108 str[size] = '\0';
109 source_ = str;
110 delete[] str;
111 return true;
112 }
113 return false;
114 }

142 int readBinaryFromFile(const char* fileName)
143 {
144 FILE * input = NULL;
145 size_t size = 0,val;
146 char* binary = NULL;
147
148 if(fopen_s(&input, fileName, "rb"))
149 {
150 return SDK_FAILURE;
151 }
152 fseek(input, 0L, SEEK_END);
153 size = ftell(input);
154 rewind(input);
155 binary = (char*)malloc(size);
156 if(binary == NULL)
157 {
158 return SDK_FAILURE;
159 }
160 val=fread(binary, sizeof(char), size, input);
161 fclose(input);
162 source_.assign(binary, size);
163 free(binary);
164 return SDK_SUCCESS;
165 }
