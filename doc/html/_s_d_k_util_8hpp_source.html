216static bool compare(const float *refData, const float *data,
217 const int length, const float epsilon = 1e-6f)
218{
219 float error = 0.0f;
220 float ref = 0.0f;
221 for(int i = 1; i < length; ++i)
222 {
223 float diff = refData[i] - data[i];
224 error += diff * diff;
225 ref += refData[i] * refData[i];
226 }
227 float normRef =::sqrtf((float) ref);
228 if (::fabs((float) ref) < 1e-7f)
229 {
230 return false;
231 }
232 float normError = ::sqrtf((float) error);
233 error = normError / normRef;
234 return error < epsilon;
235}

298static int fileToString(std::string &fileName, std::string &str)
299{
300 size_t size;
301 char* buf;
302 // Open file stream
303 std::fstream f(fileName.c_str(), (std::fstream::in | std::fstream::binary));
304 // Check if we have opened file stream
305 if (f.is_open())
306 {
307 size_t sizeFile;
308 // Find the stream size
309 f.seekg(0, std::fstream::end);
310 size = sizeFile = (size_t)f.tellg();
311 f.seekg(0, std::fstream::beg);
312 buf = new char[size + 1];
313 if (!buf)
314 {
315 f.close();
316 return SDK_FAILURE;
317 }
318 // Read file
319 f.read(buf, sizeFile);
320 f.close();
321 str[size] = '\0';
322 str = buf;
323 return SDK_SUCCESS;
324 }
325 else
326 {
327 sdk_error("Converting file to string. Cannot open file.");
328 str = "";
329 return SDK_FAILURE;
330 }
331}

459template<typename T>
460int fillRandom(
461 T * arrayPtr,
462 const int width,
463 const int height,
464 const T rangeMin,
465 const T rangeMax,
466 unsigned int seed=123)
467{
468 if(!arrayPtr)
469 {
470 sdk_error("Cannot fill array. NULL pointer.");
471 return SDK_FAILURE;
472 }
473 if(!seed)
474 {
475 seed = (unsigned int)time(NULL);
476 }
477 srand(seed);
478 double range = double(rangeMax - rangeMin) + 1.0;
479 /* random initialisation of input */
480 for(int i = 0; i < height; i++)
481 for(int j = 0; j < width; j++)
482 {
483 int index = i*width + j;
484 arrayPtr[index] = rangeMin + T(range*rand()/(RAND_MAX + 1.0));
485 }
486 return SDK_SUCCESS;
487}

547{

550 for(int i = 0; i < bytes; i++)
551 {
552 val |= val >> (1<<i);
553 }
554 val++;

606 std::string str(buffer);

664 struct Timer
665 {
666 std::string name;
667 long long _freq;
668 long long _clocks;

700 int createTimer()
701 {
702 Timer* newTimer = new Timer;
703 newTimer->_start = 0;
704 newTimer->_clocks = 0;
705#ifdef _WIN32
706 QueryPerformanceFrequency((LARGE_INTEGER*)&newTimer->_freq);
707#else
708 newTimer->_freq = (long long)1.0E3;
709#endif
710 /* Push back the address of new Timer instance created */
711 _timers.push_back(newTimer);
712 return (int)(_timers.size() - 1);
713 }
714
718 int resetTimer(int handle)
719 {
720 if(handle >= (int)_timers.size())
721 {
722 sdk_error("Cannot reset timer. Invalid handle.");
723 return -1;
724 }
725 (_timers[handle]->_start) = 0;
726 (_timers[handle]->_clocks) = 0;
727 return SDK_SUCCESS;
728 }
732 int startTimer(int handle)
733 {
734 if(handle >= (int)_timers.size())
735 {
736 sdk_error("Cannot reset timer. Invalid handle.");
737 return SDK_FAILURE;
738 }
739#ifdef _WIN32
740 QueryPerformanceCounter((LARGE_INTEGER*)&(_timers[handle]->_start));
741#else
742 struct timeval s;
743 gettimeofday(&s, 0);
744 _timers[handle]->_start = (long long)s.tv_sec * (long long)1.0E3 +
745 (long long)s.tv_usec / (long long)1.0E3;
746#endif
747 return SDK_SUCCESS;
748 }
749
753 int stopTimer(int handle)
754 {
755 long long n=0;
756 if(handle >= (int)_timers.size())
757 {
758 sdk_error("Cannot reset timer. Invalid handle.");
759 return SDK_FAILURE;
760 }
761#ifdef _WIN32
762 QueryPerformanceCounter((LARGE_INTEGER*)&(n));
763#else
764 struct timeval s;
765 gettimeofday(&s, 0);
766 n = (long long)s.tv_sec * (long long)1.0E3+ (long long)s.tv_usec /
767 (long long)1.0E3;
768#endif
769 n -= _timers[handle]->_start;
770 _timers[handle]->_start = 0;
771 _timers[handle]->_clocks += n;
772 return SDK_SUCCESS;
773 }
774
778 double readTimer(int handle)
779 {
780 if(handle >= (int)_timers.size())
781 {
782 sdk_error("Cannot read timer. Invalid handle.");
783 return SDK_FAILURE;
784 }
785 double reading = double(_timers[handle]->_clocks);
786 reading = double(reading / _timers[handle]->_freq);
787 return reading;
788 }

850 int match(char ** argv, int argc)
851 {
852 int matched = 0;
853 int shortVer = true;
854 char * arg = *argv;
855 if ( *arg == '-' && *(arg+1) == '-')
856 {
857 shortVer = false;
858 arg++;
859 }
860 else if (*arg != '-')
861 {
862 return matched;
863 }
864 arg++;
865 for (int count = 0; count < _numArgs; count++)
866 {
867 if (shortVer)
868 {
869 matched = _options[count]._sVersion.compare(arg) == 0 ? 1 : 0;
870 }
871 else
872 {
873 matched = _options[count]._lVersion.compare(arg) == 0 ? 1 : 0;
874 }
875 if (matched == 1)
876 {
877 if (_options[count]._type == CA_NO_ARGUMENT)
878 {
879 *((bool *)_options[count]._value) = true;
880 }
881 else if (_options[count]._type == CA_ARG_FLOAT)
882 {
883 if(argc > 1)
884 {
885 sscanf(*(argv+1), "%f", (float *)_options[count]._value);
886 matched++;
887 }
888 else
889 {
890 std::cout<<"Error. Missing argument for \""<<(*argv)<<"\"\n";
891 return SDK_FAILURE;
892 }
893 }
894 else if (_options[count]._type == CA_ARG_DOUBLE)
895 {
896 if(argc > 1)
897 {
898 sscanf(*(argv+1), "%lf", (double *)_options[count]._value);
899 matched++;
900 }
901 else
902 {
903 std::cout<<"Error. Missing argument for \""<<(*argv)<<"\"\n";
904 return SDK_FAILURE;
905 }
906 }
907 else if (_options[count]._type == CA_ARG_INT)
908 {
909 if(argc > 1)
910 {
911 sscanf(*(argv+1), "%d", (int *)_options[count]._value);
912 matched++;
913 }
914 else
915 {
916 std::cout<<"Error. Missing argument for \""<<(*argv)<<"\"\n";
917 return SDK_FAILURE;
918 }
919 }
920 else
921 {
922 if(argc > 1)
923 {
924 std::string * str = (std::string *)_options[count]._value;
925 str->clear();
926 str->append(*(argv+1));
927 matched++;
928 }
929 else
930 {
931 std::cout<<"Error. Missing argument for \""<<(*argv)<<"\"\n";
932 return SDK_FAILURE;
933 }
934 }
935 break;
936 }
937 }
938 return matched;
939 }

1034 int parse(char ** p_argv, int p_argc)
1035 {
1036 int matched = 0;
1037 int argc;
1038 char **argv;
1039 _argc = p_argc;
1040 _argv = p_argv;
1041 argc = p_argc;
1042 argv = p_argv;
1043 if(argc == 1)
1044 {
1045 return SDK_FAILURE;
1046 }
1047 while (argc > 0)
1048 {
1049 matched = match(argv,argc);
1050 argc -= (matched > 0 ? matched : 1);
1051 argv += (matched > 0 ? matched : 1);
1052 }
1053 return matched;
1054 }
