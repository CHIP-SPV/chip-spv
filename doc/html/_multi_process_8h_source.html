72template <typename T>
73pid_t MultiProcess<T>::SpawnProcess(bool debug_bkpt) {
74 if (num_proc_ < 0) {
75 std::cout<<"Num Process cannot be less than 1"<<std::endl;
76 return -1;
77 }
78
79 pid_t pid;
80 for (size_t proc_idx = 0; proc_idx < num_proc_; ++proc_idx) {
81 pid = fork();
82 if (pid < 0) {
83 std::cout<<"Fork Failed"<<std::endl;
84 assert(false);
85 } else if (pid == 0) {
86 //Child Process, so break
87 break;
88 }
89 }
90
91 if (debug_bkpt) {
92 DebugInfo(pid);
93 }
94
95 return pid;
96}

98template <typename T>
99bool MultiProcess<T>::CreateShmem() {
100 if (num_proc_ < 0) {
101 std::cout<<"Num Process cannot be less than 1"<<std::endl;
102 return false;
103 }
104
105 char name_template[] = "/tmp/eventXXXXX";
106 int temp_fd = mkstemp(name_template);
107 shmem_meta_obj_.shmem_name_ = name_template;
108 shmem_meta_obj_.shmem_name_.replace(0, 5, "/hip_");
109 shmem_meta_obj_.shmem_fd_ = shm_open(shmem_meta_obj_.shmem_name_.c_str(),
110 O_RDWR | O_CREAT, 0777);
111
112 if (ftruncate(shmem_meta_obj_.shmem_fd_, sizeof(ShmemMeta<T>)) != 0) {
113 std::cout<<"Cannot FTruncate "<<std::endl;
114 exit(0);
115 }
116
117 shmem_meta_obj_.shmem_ = (Shmem<T>*)mmap(0, sizeof(Shmem<T>), PROT_READ | PROT_WRITE,
118 MAP_SHARED, shmem_meta_obj_.shmem_fd_, 0);
119 memset(&shmem_meta_obj_.shmem_->handle_, 0x00, sizeof(T));
120 shmem_meta_obj_.shmem_->done_counter_ = -1;
121
122 return true;
123}
